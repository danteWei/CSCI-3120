/* Binary search tree, unique elements only */

treenode:
  int data;
  treenode *left, *right, *parent;

search( treenode *root, int data )
{
  treenode *current;
  treenode *parent;
  treenode *node;

  parent = NULL;
  current = *root;

  while ((current != NULL) && (current->data != data)) {
      parent = current;
    if (current->data > data) {
      current = current->left;
    } else {
      current = current->right;
    }
  }

  return current;
}
